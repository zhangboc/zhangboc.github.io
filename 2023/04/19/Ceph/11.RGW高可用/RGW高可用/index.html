2. keepalived安装配置

2.1 安装软件

yum -y install keepalived 

分别在两个节点node-1和node-2上执⾏安装即可。

2.2 配置VIP

master节点配置(node-1)


[root@node-1 my-cluster]#  cd /etc/keepalived/

[root@node-1 keepalived]# ls
keepalived.conf
[root@node-1 keepalived]# cat keepalived.conf 
! Configuration File for keepalived

global_defs {
   notification_email {
     acassen@firewall.loc
     failover@firewall.loc
     sysadmin@firewall.loc
   }
   notification_email_from Alexandre.Cassen@firewall.loc
   smtp_server 192.168.200.1
   smtp_connect_timeout 30
   router_id LVS_DEVEL
   vrrp_skip_check_adv_addr

   vrrp_garp_interval 0
   vrrp_gna_interval 0
}

vrrp_script chk_haproxy {
 script "killall -0 haproxy"
 interval 2
 weight -2
}
vrrp_instance RGW {
    state MASTER
    interface eth0  #有可能需要修改网卡
    virtual_router_id 51
    priority 100
    advert_int 1
    authentication {
        auth_type PASS
        auth_pass 1111
    }
    virtual_ipaddress {
        192.168.187.200/24 #需要修改VIP地址
    }
    track_script {
    chk_haproxy
    }
}
